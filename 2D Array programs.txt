// To print sum of daigonals in (3/3) matrix
#include<stdio.h>
int main()
{  
    int x[3][3],i,j,sum=0;
    for(i=0;i<3;i++)
     { 
         for(j=0;j<3;j++)
          {
              scanf("%d",&x[i][j]);
              
          }
     }  
     for(i=0;i<3;i++)
{
    for(j=0;j<3;j++)
    {
        if(i==j)
        {
            sum+=x[i][j];
        }
    }
}
printf("%d",sum);
return 0;
}
